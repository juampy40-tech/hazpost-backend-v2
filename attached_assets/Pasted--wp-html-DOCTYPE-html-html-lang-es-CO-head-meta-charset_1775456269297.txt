        window._pendingPixelEvents = window._pendingPixelEvents || [];

    if (typeof gtag !== 'undefined') {
        gtag('event', eventName.toLowerCase(), enhancedData);
    }
}

    function validarEntrada(valor, min, max, mensaje, campoId) {
        // Eliminar mensajes de error existentes
        const erroresExistentes = document.querySelectorAll('.error-message');
        erroresExistentes.forEach(error => {

    // CÁLCULO DE PROYECCIÓN COMPRA
    function calcularProyeccionCompra(costoSistema, beneficioTributarioTotal, tamanoSistema, consumoMensual, costoKwhActual) {
        const tbody = document.getElementById('compra-proyeccion-body');
        tbody.innerHTML = '';
        
        let recuperacionInversion = -costoSistema;
        let anioRecuperacion = 0;
        let mantenimientoAnual = 0;
        let ahorroAcumulado = 0;
        let totalAcumulado = 0;
        let generacionTotal = 0;
        
        const labels = [];
        const ahorrosData = [];
        const inversionData = [];
        
        const generacionAnualEstimada = consumoMensual * 12 / 0.84 * 0.86;
        
        for (let ano = 0; ano <= 25; ano++) {
            const row = document.createElement('tr');
            
            if (ano === 0) {
                row.innerHTML = `
                    <td>${ano}</td>
                    <td>-</td>
                    <td>-</td>
                    <td>-</td>
                    <td>-</td>
                    <td>${formatoMoneda(recuperacionInversion)}</td>
                `;
                tbody.appendChild(row);
                labels.push(ano);
                ahorrosData.push(0);
                inversionData.push(recuperacionInversion);
                continue;
            }
            
            const eficiencia = eficienciaPorAnoCompra[ano];
            const generacionAnual = generacionAnualEstimada * eficiencia;
            generacionTotal += generacionAnual;
            
            const costoKwhAnual = costoKwhActual * Math.pow(1.06, ano-1);
            const ahorroAnual = generacionAnual * costoKwhAnual;
            ahorroAcumulado += ahorroAnual;
            
            if (ano === 1) {
                document.getElementById('compra-ahorro-primer-ano').textContent = formatoMoneda(ahorroAnual);
            }
            
            if (ano >= 2) {
                mantenimientoAnual = costoSistema * 0.01 * Math.pow(1.10, ano-2);
            } else {
                mantenimientoAnual = 0;
            }
            
            let beneficioTributarioAnual = 0;
            if (ano === 2) beneficioTributarioAnual = beneficioTributarioTotal * 0.5556;
            else if (ano === 3 || ano === 4) beneficioTributarioAnual = beneficioTributarioTotal * 0.2222;
            
            if (ano === 1) {
                recuperacionInversion += ahorroAnual;
            } else {
                recuperacionInversion += ahorroAnual - mantenimientoAnual + beneficioTributarioAnual;
            }
            
            if (recuperacionInversion >= 0 && anioRecuperacion === 0) {
                anioRecuperacion = ano;
                document.getElementById('compra-retorno-inversion').textContent = `Aprox. ${ano} años`;
            }
            
            if (ano === 25) totalAcumulado = recuperacionInversion;
            
            row.innerHTML = `
                <td>${ano}</td>
                <td>${formatoNumero(generacionAnual.toFixed(2))}</td>
                <td>${formatoMoneda(ahorroAnual)}</td>
                <td>${ano >= 2 ? formatoMoneda(mantenimientoAnual) : '-'}</td>
                <td>${beneficioTributarioAnual > 0 ? formatoMoneda(beneficioTributarioAnual) : '-'}</td>
                <td>${formatoMoneda(recuperacionInversion)}</td>
            `;
            
            if (ano === anioRecuperacion) row.classList.add('recuperado');
            
            tbody.appendChild(row);
            labels.push(ano);
            ahorrosData.push(ahorroAnual);
            inversionData.push(recuperacionInversion);
        }
        
        document.getElementById('compra-ahorro-acumulado').textContent = formatoMoneda(totalAcumulado + costoSistema);
        
        const totalContainer = document.getElementById('compra-total-acumulado-container');
        totalContainer.innerHTML = `TOTAL ACUMULADO EN 25 AÑOS: <strong>${formatoMoneda(totalAcumulado)}</strong>`;
        totalContainer.style.display = 'block';
        
        document.getElementById('compra-utilidad-total').textContent = formatoMoneda(totalAcumulado + costoSistema);
        mostrarBeneficiosAmbientalesCompra(generacionTotal);
        
        const totalRow = document.createElement('tr');
        totalRow.classList.add('fila-total');
        totalRow.innerHTML = `
            <td><strong>25 AÑOS</strong></td>
            <td><strong>${formatoNumero(generacionTotal.toFixed(2))}</strong></td>
            <td><strong>${formatoMoneda(ahorroAcumulado)}</strong></td>
            <td><strong>${formatoMoneda(costoSistema * 0.01 * (Math.pow(1.10, 23) - 1) / 0.10)}</strong></td>
            <td><strong>${formatoMoneda(beneficioTributarioTotal)}</strong></td>
            <td><strong>${formatoMoneda(totalAcumulado)}</strong></td>
        `;
        tbody.appendChild(totalRow);
        
        crearGraficosCompra(labels, ahorrosData, inversionData);
    }

    // CALCULADORA FINANCIERA COMPRA
    function calcularFinanciamientoCompra() {
        const montoPrestamo = monedaANumero(document.getElementById('compra-monto-prestamo').value);
        const tasaInteresAnual = parseFloat(document.getElementById('compra-tasa-interes').value) || 0;
        const plazoAnios = parseFloat(document.getElementById('compra-plazo-prestamo').value) || 0;
        
        if (montoPrestamo <= 0 || tasaInteresAnual <= 0 || plazoAnios <= 0) {
            alert("Por favor complete todos los campos con valores válidos.");
            return;
        }

        const tasaMensual = tasaInteresAnual / 100 / 12;
        const plazoMeses = plazoAnios * 12;
        
        // Calcular cuota mensual
        const cuotaMensual = montoPrestamo * 
            (tasaMensual * Math.pow(1 + tasaMensual, plazoMeses)) / 
            (Math.pow(1 + tasaMensual, plazoMeses) - 1);
        
        const totalPagar = cuotaMensual * plazoMeses;
        const interesesTotales = totalPagar - montoPrestamo;
        
        // Mostrar resultados
        document.getElementById('compra-cuota-mensual').textContent = formatoMoneda(cuotaMensual);
        document.getElementById('compra-total-pagar').textContent = formatoMoneda(totalPagar);
        document.getElementById('compra-intereses-totales').textContent = formatoMoneda(interesesTotales);
        
        // Actualizar comparación financiera
        document.getElementById('compra-cuota-prestamo-inicial').textContent = formatoMoneda(cuotaMensual);
        document.getElementById('compra-total-prestamo').textContent = formatoMoneda(totalPagar);
        
        // Comparación con ahorro energético
        if (ahorroMensualGlobalCompra > 0) {
            const comparacion = ahorroMensualGlobalCompra >= cuotaMensual ? 
                "¡Tu ahorro energético cubre la cuota del crédito!" : 
                "Con el ahorro energético cubres el " + ((ahorroMensualGlobalCompra / cuotaMensual) * 100).toFixed(0) + "% de la cuota";
            
            document.getElementById('compra-comparacion-ahorro').textContent = comparacion;
        }
        
        document.getElementById('compra-resultado-financiamiento').style.display = 'block';
        
        // Trackear evento de cálculo de financiamiento
        trackMetaPixelEvent('CalculateLoan', {
            loan_amount: montoPrestamo,
            interest_rate: tasaInteresAnual,
            loan_term: plazoAnios,
            monthly_payment: cuotaMensual,
            total_payment: totalPagar
        });
    }

            currency: 'COP'
        });
        
        // Guardar datos en localStorage
        guardarDatosFormulario('ppa', {
            kwhPromedio: kwhPromedio,
            valorFactura: valorFactura,
            kwhFacturados: kwhFacturados
        });
        
        // Calcular tamaño del sistema
        const numPanelesTemp = (kwhPromedio / 4 / 30 / 0.84) / 0.645;
        const numPaneles = Math.ceil(numPanelesTemp);
        const tamanoSistema = numPaneles * 0.645;
        
        // Verificar tamaño mínimo del sistema (10 kW)
        if (tamanoSistema < 10) {
            alert('Lo sentimos, este programa solo aplica para sistemas mayores a 10 kW. Tu sistema calculado es de ' + tamanoSistema.toFixed(2) + ' kW. Por favor contáctanos para otras opciones.');
            return;
        }
        
        // Calcular producción mensual estimada
        const produccionMensual = tamanoSistema * 4 * 30 * 0.84;
        
        // Calcular tarifa actual por kWh (20% menos que lo que paga actualmente)
        const tarifaActual = (valorFactura / kwhFacturados) * 0.2;
        
        // Calcular ahorros
        let ahorroAcumulado = 0;
        let ahorroPrimerAno = 0;
        let tarifaActualCalculo = tarifaActual;
        const resultadosAnuales = [];
        let generacionTotal = 0;
        
        for (let ano = 0; ano < 15; ano++) {
            // Aumentar tarifa 6% cada año (excepto el primero)
            if (ano > 0) {
                tarifaActualCalculo *= 1.06;
            }
            
            // Calcular producción anual considerando eficiencia
            const produccionAnual = produccionMensual * 12 * eficienciaPorAnoPPA[ano];
            generacionTotal += produccionAnual;
            
            // Calcular ahorro anual
            const ahorroAnual = produccionAnual * tarifaActualCalculo;
            
            if (ano === 0) {
                ahorroPrimerAno = ahorroAnual;
            }
            
            ahorroAcumulado += ahorroAnual;
            
            resultadosAnuales.push({
                ano: ano + 1,
                produccion: produccionAnual,
                ahorroAnual: ahorroAnual,
                ahorroAcumulado: ahorroAcumulado
            });
        }
        
        // Mostrar resultados principales
        document.getElementById('ppa-tamano-sistema').textContent = tamanoSistema.toFixed(2);
        document.getElementById('ppa-numero-paneles').textContent = numPaneles;
        document.getElementById('ppa-area-requerida').textContent = (tamanoSistema * 5.2).toFixed(2);
        document.getElementById('ppa-produccion-mensual').textContent = produccionMensual.toFixed(2);
        
        // Mostrar resultados financieros
        document.getElementById('ppa-ahorro-primer-mes').textContent = formatoMoneda(ahorroPrimerAno / 12);
        document.getElementById('ppa-ahorro-primer-ano').textContent = formatoMoneda(ahorroPrimerAno);
        document.getElementById('ppa-ahorro-total').textContent = formatoMoneda(ahorroAcumulado);
        document.getElementById('ppa-ahorro-total-destacado').textContent = formatoMoneda(ahorroAcumulado);
        document.getElementById('ppa-ahorro-total-destacado2').textContent = formatoMoneda(ahorroAcumulado);
        document.getElementById('ppa-tarifa-aplicada').textContent = formatoMoneda(tarifaActual);
        
        // Calcular y mostrar beneficios ambientales
        mostrarBeneficiosAmbientalesPPA(generacionTotal);
        
        // Llenar tabla detallada
        const tablaDetalle = document.getElementById('ppa-detalle-anual');
        tablaDetalle.innerHTML = '';
        
        resultadosAnuales.forEach(item => {
            const fila = document.createElement('tr');
            fila.innerHTML = `
                <td>${item.ano}</td>
                <td>${formatoNumero(item.produccion)}</td>
                <td>${formatoMoneda(item.ahorroAnual)}</td>
                <td>${formatoMoneda(item.ahorroAcumulado)}</td>
            `;
            tablaDetalle.appendChild(fila);
        });
        
        // Mostrar resultados
        document.getElementById('ppa-resultado').style.display = 'block';
        
        // Trackear evento de cálculo completado
trackCalculationEvent('ppa', {
    kwhPromedio: kwhPromedio,
    valorFactura: valorFactura,
    tamanoSistema: tamanoSistema.toFixed(2),
    costoSistema: 0, // PPA no tiene costo inicial
    ahorroEstimado: ahorroAcumulado
});

// Trackear vista de resultados
trackResultsView('ppa', {
    system_size: tamanoSistema.toFixed(2),
    estimated_savings: ahorroAcumulado,
    contract_years: 15
});
        
        // Desplazarse a los resultados
        document.getElementById('ppa-resultado').scrollIntoView({behavior: 'smooth'});
    }
    
    // BENEFICIOS AMBIENTALES PPA
    function mostrarBeneficiosAmbientalesPPA(generacionTotal) {
        // Cálculos ambientales
        const co2Evitado = (generacionTotal * FACTORES_AMBIENTALES_PPA.CO2_POR_KWH) / 1000;
        const arbolesEquivalentes = (generacionTotal * FACTORES_AMBIENTALES_PPA.ARBOLES_POR_KWH) / 1000;
        const gasolinaEquivalente = generacionTotal * FACTORES_AMBIENTALES_PPA.GASOLINA_POR_KWH;
        const carrosEquivalentes = co2Evitado / FACTORES_AMBIENTALES_PPA.CO2_CARRO_ANUAL;
        
        // Mostrar resultados
        document.getElementById('ppa-arboles-salvados').textContent = formatoNumero(arbolesEquivalentes, 0);
        document.getElementById('ppa-co2-evitar').textContent = formatoNumero(co2Evitado, 1);
        document.getElementById('ppa-combustible-equivalente').textContent = formatoNumero(gasolinaEquivalente, 0);
        document.getElementById('ppa-carros-vial').textContent = formatoNumero(carrosEquivalentes, 1);
    }

    // Función principal para manejar la descarga de PDF
    async function descargarPDF() {
        // Determinar qué pestaña está activa
        const tabActiva = document.querySelector('.tab.active').textContent.includes('PPA') ? 'ppa' : 'compra';
        
        // Mostrar mensaje de carga
        const btn = event.target;
        btn.textContent = 'Generando PDF...';
        btn.disabled = true;
        
        try {
            // Cargar librerías necesarias
            await cargarLibreriasPDF();
            
            // Generar PDF según la pestaña activa
            if (tabActiva === 'compra') {
                await generarPDFCompra();
            } else {
                await generarPDFPPA();
            }
        } catch (error) {
            console.error('Error al generar PDF:', error);
            alert('Ocurrió un error al generar el PDF. Por favor intente nuevamente.');
        } finally {
            // Restaurar botón
            btn.textContent = 'DESCARGAR PRESENTACIÓN';
            btn.disabled = false;
        }
    }

    // Función para cargar las librerías necesarias
    async function cargarLibreriasPDF() {
        if (typeof jsPDF === 'undefined') {
            await loadScript('https://cdnjs.cloudflare.com/ajax/libs/jspdf/2.5.1/jspdf.umd.min.js');
        }
        
        if (typeof html2canvas === 'undefined') {
            await loadScript('https://cdnjs.cloudflare.com/ajax/libs/html2canvas/1.4.1/html2canvas.min.js');
        }
    }

    // Función auxiliar para cargar scripts dinámicamente
    function loadScript(src) {
        return new Promise((resolve, reject) => {
            const script = document.createElement('script');
            script.src = src;
            script.onload = resolve;
            script.onerror = reject;
            document.head.appendChild(script);
        });
    }
